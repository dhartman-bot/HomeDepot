Example: Modernizing Legacy POS Inventory Check

Legacy COBOL (1990s)

IDENTIFICATION DIVISION.
PROGRAM-ID. INVCHECK.
DATA DIVISION.
WORKING-STORAGE SECTION.
01  STORE-NUM      PIC 9(4).
01  ITEM-SKU       PIC X(12).
01  QTY-ON-HAND    PIC 9(6).
01  QTY-RESERVED   PIC 9(6).
01  QTY-AVAILABLE  PIC 9(6).

PROCEDURE DIVISION.
    ACCEPT STORE-NUM.
    ACCEPT ITEM-SKU.
    CALL 'DB2QUERY' USING
         STORE-NUM ITEM-SKU
         QTY-ON-HAND QTY-RESERVED.
    COMPUTE QTY-AVAILABLE =
         QTY-ON-HAND - QTY-RESERVED.
    DISPLAY QTY-AVAILABLE.
    STOP RUN.

Modern Node.js API (Claude Generated)

// Generated by Claude Code
const express = require('express');
const router = express.Router();

/**
 * Get available inventory for a SKU
 * @route GET /api/inventory/check
 */
router.get('/check', async (req, res) => {
  const { storeNum, sku } = req.query;

  try {
    const inventory = await db.query(`
      SELECT
        qty_on_hand,
        qty_reserved,
        qty_on_hand - qty_reserved
          AS qty_available
      FROM inventory
      WHERE store_num = $1 AND sku = $2
    `, [storeNum, sku]);

    res.json(inventory.rows[0]);
  } catch (error) {
    res.status(500).json({
      error: 'Inventory check failed'
    });
  }
});

module.exports = router;
